while문
			while(조건식){
				실행문;

			}


조건식은 생략불가(무한루프 만드려면 조건식에 true를 입력)

do while문		반드시 한번은 실행(콘솔 프로그램)

			do{
				실행문;
	
			}while(조건식);

break			반복문을 빠져나갈 때 사용(if문과 함께)

continue			특정 위치로 이동(for문은 증감식, while문은 조건식으로 이동)
			(if문과 함께)

중첩반복문		반복문 안에 반복문

향상된 for문		배열 / 컬렉션을 이용해서 반복할 때 사용
			전체 탐색 할 때 사용

			for( 자료형 변수명 : 배열/컬렉션 ) {
				실행문;
			}
